q1/ trouver le/les livres empruntés par quelqu'un(code barre)

SELECT titre,editeur,annee,isbn FROM LIVRE
WHERE isbn=(
	SELECT isbn from EMPRUNT
	WHERE code_barre=421921003090881
)

q2/ trouver celui/ceux qui ont empruntés un livre(isbn)

SELECT nom,prenom,code_barre FROM USAGER
WHERE code_barre = (
	SELECT code_barre FROM EMPRUNT
	WHERE isbn="978-2081358881"
)

q3/ trouver qui n'a pas rendu son livre alors qu'il devait être rendu le 17/09/2020

SELECT nom,prenom,code_barre FROM USAGER
WHERE code_barre = (
	SELECT code_barre FROM EMPRUNT
	WHERE retour > "2020-09-17"
)

q4/ trouver les livres avec "aventure" dans le titre

SELECT titre,editeur,annee,isbn FROM LIVRE
WHERE titre LIKE "%aventure%"
